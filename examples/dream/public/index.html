let () =
  let port =
    match Sys.getenv_opt "PORT" with
    | Some p -> int_of_string p
    | None -> 3000
  in
  Dream.run ~port
  @@ Dream.logger
  @@ Dream.router [
    Dream.get "/" (fun _ ->
      Dream.json {|{"message": "OCaml on Vercel!"}|});
  ]
