"use client"

import type React from "react"

import { useState } from "react"
import { CalendarIcon, CheckCircle } from 'lucide-react'
import { format } from "date-fns"

import { Button } from "@/components/ui/button"
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { Textarea } from "@/components/ui/textarea"
import { Calendar } from "@/components/ui/calendar"
import { Popover, PopoverContent, PopoverTrigger } from "@/components/ui/popover"
import { cn } from "@/lib/utils"

export default function CoachingForm() {
  const [date, setDate] = useState<Date>()
  const [isSubmitting, setIsSubmitting] = useState(false)
  const [isSubmitted, setIsSubmitted] = useState(false)

  const handleSubmit = async (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault()
    setIsSubmitting(true)

    const formData = new FormData(e.currentTarget)
    const data = Object.fromEntries(formData.entries())

    // Add the date to the data
    if (date) {
      data.sessionDate = format(date, "yyyy-MM-dd")
    }

    try {
      const response = await fetch(
        "https://script.google.com/macros/s/AKfycbwLpDd5iUaLbZkXpMhpnlMhtKzBKO4NkB1XT1G8aeeeaieAY8u1RmjwqVC5TLTxkEP6/exec",
        {
          method: "POST",
          body: JSON.stringify(data),
          headers: {
            "Content-Type": "application/json",
          },
        },
      )

      if (response.ok) {
        setIsSubmitted(true)
        // Reset form
        e.currentTarget.reset()
        setDate(undefined)

        // Hide success message after 5 seconds
        setTimeout(() => {
          setIsSubmitted(false)
        }, 5000)
      }
    } catch (error) {
      console.error("Error:", error)
      alert("Error submitting form. Please try again.")
    } finally {
      setIsSubmitting(false)
    }
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100">
      <header className="bg-blue-600 text-white py-8">
        <div className="container mx-auto px-4">
          <h1 className="text-3xl font-bold text-center">Coaching Form</h1>
          <p className="text-center mt-2 text-blue-100">Professional Development & Training Documentation</p>
        </div>
      </header>

      <div className="container mx-auto px-4 py-8">
        <div className="max-w-4xl mx-auto">
          {isSubmitted && (
            <div className="mb-6 p-4 bg-green-50 border border-green-200 rounded-lg flex items-center gap-2 text-green-800">
              <CheckCircle className="h-5 w-5" />
              <span className="font-medium">Form submitted successfully!</span>
            </div>
          )}

          <form onSubmit={handleSubmit} className="space-y-8">
            {/* Agent Info Section */}
            <Card>
              <CardHeader>
                <CardTitle className="text-blue-600">Agent Information</CardTitle>
                <CardDescription>
                  Please provide the basic information about the coaching session and participant.
                </CardDescription>
              </CardHeader>
              <CardContent className="space-y-6">
                <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                  <div className="space-y-2">
                    <Label htmlFor="typeOfCoaching">Type of Coaching *</Label>
                    <Input
                      id="typeOfCoaching"
                      name="typeOfCoaching"
                      required
                      placeholder="e.g., Performance, Skills Development"
                    />
                  </div>
                  <div className="space-y-2">
                    <Label htmlFor="businessUnit">Business Unit</Label>
                    <Input id="businessUnit" name="businessUnit" placeholder="Enter business unit" />
                  </div>
                  <div className="space-y-2">
                    <Label htmlFor="teamName">Team Name</Label>
                    <Input id="teamName" name="teamName" placeholder="Enter team name" />
                  </div>
                  <div className="space-y-2">
                    <Label htmlFor="agentName">Agent's Name</Label>
                    <Input id="agentName" name="agentName" placeholder="Enter agent's full name" />
                  </div>
                  <div className="space-y-2">
                    <Label>Date of Session</Label>
                    <Popover>
                      <PopoverTrigger asChild>
                        <Button
                          variant="outline"
                          className={cn("w-full justify-start text-left font-normal", !date && "text-muted-foreground")}
                        >
                          <CalendarIcon className="mr-2 h-4 w-4" />
                          {date ? format(date, "PPP") : "Pick a date"}
                        </Button>
                      </PopoverTrigger>
                      <PopoverContent className="w-auto p-0">
                        <Calendar mode="single" selected={date} onSelect={setDate} initialFocus />
                      </PopoverContent>
                    </Popover>
                  </div>
                  <div className="space-y-2">
                    <Label htmlFor="referenceNumber">Interaction Reference Number</Label>
                    <Input id="referenceNumber" name="referenceNumber" placeholder="Enter reference number" />
                  </div>
                  <div className="space-y-2">
                    <Label htmlFor="email">Coachee Email Address</Label>
                    <Input id="email" name="email" type="email" placeholder="coachee@company.com" />
                  </div>
                  <div className="space-y-2">
                    <Label htmlFor="employeeId">Employee ID</Label>
                    <Input id="employeeId" name="employeeId" placeholder="Enter employee ID" />
                  </div>
                  <div className="space-y-2">
                    <Label htmlFor="logonId">Log on ID</Label>
                    <Input id="logonId" name="logonId" placeholder="Enter logon ID" />
                  </div>
                </div>
              </CardContent>
            </Card>

            {/* Coaching Session Details */}
            <Card>
              <CardHeader>
                <CardTitle className="text-blue-600">Off the Phone Coaching Session</CardTitle>
                <CardDescription>Document the details and outcomes of the coaching session.</CardDescription>
              </CardHeader>
              <CardContent className="space-y-6">
                <div className="space-y-2">
                  <Label htmlFor="topic">What is the topic of today's coaching session?</Label>
                  <Textarea
                    id="topic"
                    name="topic"
                    rows={3}
                    placeholder="Describe the main topic or focus area for this coaching session..."
                  />
                </div>

                <div className="space-y-2">
                  <Label htmlFor="coachedBefore">Has the coachee been previously coached on this topic?</Label>
                  <Textarea
                    id="coachedBefore"
                    name="coachedBefore"
                    rows={2}
                    placeholder="Provide details about any previous coaching on this topic..."
                  />
                </div>

                <div className="space-y-2">
                  <Label htmlFor="positives">Mention positive outcomes or what went well?</Label>
                  <Textarea
                    id="positives"
                    name="positives"
                    rows={3}
                    placeholder="Highlight the positive aspects, achievements, or improvements observed..."
                  />
                </div>

                <div className="space-y-2">
                  <Label htmlFor="goal">What is the goal for this session?</Label>
                  <Textarea
                    id="goal"
                    name="goal"
                    rows={3}
                    placeholder="Define the specific objectives and expected outcomes for this coaching session..."
                  />
                </div>

                <div className="space-y-2">
                  <Label htmlFor="support">As the Coach, how will you be supporting achieving this goal?</Label>
                  <Textarea
                    id="support"
                    name="support"
                    rows={3}
                    placeholder="Describe your coaching approach, resources, and support strategies..."
                  />
                </div>
              </CardContent>
            </Card>

            <div className="flex justify-center">
              <Button type="submit" size="lg" disabled={isSubmitting} className="bg-blue-600 hover:bg-blue-700 px-8">
                {isSubmitting ? "Submitting..." : "Submit Coaching Form"}
              </Button>
            </div>
          </form>
        </div>
      </div>
    </div>
  )
}
